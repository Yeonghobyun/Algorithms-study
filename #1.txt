#include <iostream>
#include <stack>
#include <string>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	stack<int> s;
	string paren;
	cin >> paren;

	int ans = 0;

	for (int i = 0; i < paren.size(); i++) {
		if (paren[i] == '(')
			s.push(i);

		else if(paren[i] == ')' && !s.empty()) {
			int temp = s.top();
			s.pop();
			
			if (i - temp == 1)
				ans += s.size();
			else
				ans += 1;
		}
	}
	cout << ans;

}
